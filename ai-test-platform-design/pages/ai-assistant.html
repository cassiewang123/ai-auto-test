import requests

def test_user_register_normal():
    """正常注册"""
    resp = requests.post("/api/v1/users", json={
        "username": "newuser",
        "email": "new@example.com",
        "password": "Valid123!"
    })
    assert resp.status_code == 201
    assert resp.json()["code"] == 0

def test_user_register_duplicate():
    """重复用户名"""
    resp = requests.post("/api/v1/users", json={
        "username": "existinguser",
        "email": "dup@example.com",
        "password": "Valid123!"
    })
    assert resp.status_code == 409
    assert "已存在" in resp.json()["message"]

def test_user_register_invalid_email():
    """无效邮箱"""
    resp = requests.post("/api/v1/users", json={
        "username": "testuser",
        "email": "not-an-email",
        "password": "Valid123!"
    })
    assert resp.status_code == 422
